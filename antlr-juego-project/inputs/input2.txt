juego aventura2 {
    escena inicio {
        mostrar "Bienvenido a la aventura 2";
        mostrar "Ingresa tu nombre (Alice, Bob, o cualquier otro)";
        leer nombre;
        si nombre == "Alice" saltar bosque;
        si nombre == "Bob" saltar ciudad;
    }
    escena camino {
        mostrar "Has tomado un camino desconocido";
        mostrar "Parece que no eres ni Alice ni Bob";
        mostrar "Hola, viajero misterioso";
        saltar final;
    }
    escena bosque {
        mostrar "Estás en el bosque encantado";
        mostrar "Alice, encuentras una cabaña misteriosa";
        saltar final;
    }
    escena ciudad {
        mostrar "Has llegado a la ciudad";
        mostrar "Bob, ves un mercado bullicioso";
        saltar final;
    }
    escena final {
        mostrar "Fin de la aventura";
        mostrar "Gracias por jugar";
    }
}
